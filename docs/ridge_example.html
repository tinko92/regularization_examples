#include <iostream>
#include <random>
#include <Eigen/Dense>
#include <vector>

const int N = 10;
const int p = 3;
const int runs = 1'000'000;

std::pair<Eigen::Matrix<double,N,p>,Eigen::Matrix<double,N,1>> generate_sample() {
    std::default_random_engine generator(std::random_device{}());
    std::normal_distribution hidden_dist(0.0,10.0);
    std::normal_distribution input_noise(0.0,0.01);
    std::normal_distribution output_noise(0.0,1);
    Eigen::Matrix X;
    Eigen::Matrix y;
    for(int i=0;i<N;++i) {
        auto hidden1 = hidden_dist(generator);
        auto hidden2 = hidden_dist(generator);
        auto x1 = hidden1 - 2*hidden2 + input_noise(generator);
        auto x2 = -hidden1 + 1.5*hidden2 + input_noise(generator);
        auto x3 = hidden1 + hidden2 + input_noise(generator);
        X.row(i) << x1,x2,x3;
        y.row(i) << 2*x1-x2+x3 +output_noise(generator);
    }
    return std::make_pair(X,y);
};

int main() {
    for(auto lambda : {0.0,0.01,0.1,1.0,10.0,100.0}) {
        double ls_test = 0.0, ridge_test = 0.0;
        double ls_training = 0.0, ridge_training = 0.0;
	double df = 0.0;
        for (int j = 0; j < runs; ++j) {
            auto[X, y] = generate_sample();
            Eigen::Matrix b_ls = (X.transpose() * X).ldlt().solve(X.transpose() * y);
            Eigen::Matrix b_ridge = (X.transpose() * X + lambda *
            Eigen::Matrix::Identity()).ldlt().solve(X.transpose() * y);
            auto[X_test, y_test] = generate_sample();
            ls_training += (X * b_ls - y).squaredNorm();
            ridge_training += (X * b_ridge - y).squaredNorm();
            ls_test += (X_test * b_ls - y_test).squaredNorm();
            ridge_test += (X_test * b_ridge - y_test).squaredNorm();
	    df += (X*(X.transpose()*X+lambda*Eigen::Matrix::Identity()).inverse()*X.transpose()).trace();
        }
        std::cout << "lambda: \t\t\t" << lambda << "\n";
        std::cout << "ls_training: \t\t"<< ls_training / runs << "\n";
        std::cout << "ridge_training: \t" << ridge_training / runs << "\n";
        std::cout << "ls_test: \t\t\t" << ls_test / runs << "\n";
        std::cout << "ridge_test: \t\t" << ridge_test / runs << "\n";
	std::cout << "df(lambda): \t\t" << df/runs << "\n";
    }
}

This will perform 1,000,000 runs of the example and print the average errors and degree of freedom to standard output. A compiler with C++17 support and Eigen3 are required to build. On most unix systems with the dependencies installed to default directories it should build with 

g++ main.cpp -o ridge_test -std=c++17 -I /usr/include/eigen3
